
CREATE TABLE operators (
            id TEXT PRIMARY KEY,
            name TEXT NOT NULL,
            password TEXT NOT NULL,
            contact1 TEXT,
            contact2 TEXT,
            address1 TEXT,
            address2 TEXT,
            email1 TEXT,
            email2 TEXT,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
            updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        , acteur_type_id TEXT REFERENCES acteurs_type(id), project_id TEXT REFERENCES projects(id))

CREATE TABLE action_logs (
            id TEXT PRIMARY KEY,
            user_id TEXT,
            action TEXT NOT NULL,
            target_table TEXT NOT NULL,
            target_record_id TEXT,
            action_timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
            FOREIGN KEY (user_id) REFERENCES users(id)
        )

CREATE TABLE roles (
            id TEXT PRIMARY KEY,
            role_name TEXT NOT NULL,
            description TEXT
        )

CREATE TRIGGER update_user_timestamp 
        AFTER UPDATE ON users
        BEGIN
            UPDATE users SET updated_at = CURRENT_TIMESTAMP WHERE id = NEW.id;
        END

CREATE TRIGGER update_operator_timestamp 
        AFTER UPDATE ON operators
        BEGIN
            UPDATE operators SET updated_at = CURRENT_TIMESTAMP WHERE id = NEW.id;
        END

CREATE TABLE employee_locations (
    id TEXT PRIMARY KEY,
    employee_id TEXT NOT NULL,
    contract_id TEXT NOT NULL,
    region_id TEXT NOT NULL,
    departement_id TEXT NOT NULL,
    sous_prefecture_id TEXT NOT NULL,
    date_debut DATE NOT NULL,
    date_fin DATE,  -- NULL si c'est la localisation actuelle
    motif_changement TEXT,  -- Raison du changement de localisation
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME,
    deleted_at DATETIME,
    FOREIGN KEY (employee_id) REFERENCES employees(id),
    FOREIGN KEY (contract_id) REFERENCES contracts(id),
    FOREIGN KEY (region_id) REFERENCES regions(id),
    FOREIGN KEY (departement_id) REFERENCES departements(id),
    FOREIGN KEY (sous_prefecture_id) REFERENCES sous_prefectures(id)
)

CREATE INDEX idx_employee_locations_employee ON employee_locations(employee_id)

CREATE INDEX idx_employee_locations_contract ON employee_locations(contract_id)

CREATE INDEX idx_employee_locations_dates ON employee_locations(date_debut, date_fin)

CREATE TRIGGER update_employee_location_timestamp 
AFTER UPDATE ON employee_locations
BEGIN
    UPDATE employee_locations SET updated_at = CURRENT_TIMESTAMP WHERE id = NEW.id;
END

CREATE TRIGGER check_employee_location_dates
BEFORE INSERT ON employee_locations
BEGIN
    SELECT CASE
        WHEN EXISTS (
            SELECT 1 FROM employee_locations
            WHERE employee_id = NEW.employee_id
            AND (
                (NEW.date_debut BETWEEN date_debut AND COALESCE(date_fin, '9999-12-31'))
                OR (COALESCE(NEW.date_fin, '9999-12-31') BETWEEN date_debut AND COALESCE(date_fin, '9999-12-31'))
                OR (date_debut BETWEEN NEW.date_debut AND COALESCE(NEW.date_fin, '9999-12-31'))
            )
            AND deleted_at IS NULL
        )
        THEN RAISE(ABORT, 'Les dates de localisation se chevauchent pour cet employé')
    END;
END

CREATE TRIGGER check_employee_location_date_order
BEFORE INSERT ON employee_locations
BEGIN
    SELECT CASE
        WHEN NEW.date_fin IS NOT NULL AND NEW.date_debut >= NEW.date_fin
        THEN RAISE(ABORT, 'La date de début doit être antérieure à la date de fin')
    END;
END

CREATE TRIGGER check_employee_location_contract_dates
BEFORE INSERT ON employee_locations
BEGIN
    SELECT CASE
        WHEN EXISTS (
            SELECT 1 FROM contracts
            WHERE id = NEW.contract_id
            AND (
                NEW.date_debut < start_date
                OR (NEW.date_fin IS NOT NULL AND NEW.date_fin > COALESCE(end_date, '9999-12-31'))
            )
        )
        THEN RAISE(ABORT, 'Les dates de localisation doivent être comprises dans la période du contrat')
    END;
END

            start_date DATE NOT NULL,

            updated_at DATETIME,
            deleted_at DATETIME,
            FOREIGN KEY (employee_id) REFERENCES employees(id)
        )
